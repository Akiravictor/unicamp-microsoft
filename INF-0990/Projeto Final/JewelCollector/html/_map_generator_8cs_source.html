281 private bool MapIsPlayeble(string[,] grid)
282 {

285
286 //Locally creates a jewel list
287 foreach(var j in Jewels)

289 jewelsLocal.Add(j);

292 //BFS Algorithm
293 queue.Enqueue(new Axis(0,0));
294 grid[0, 0] = "xx";
295
296 while(queue.Count > 0)
297 {
298 var node = queue.Dequeue();
299 var neighbors = Neighbors(node.X, node.Y, grid);
300
301 foreach (var neighbor in neighbors)
302 {
303 if (grid[neighbor.X, neighbor.Y] != "xx" && grid[neighbor.X, neighbor.Y] != "oo")
304 {
305 if (grid[neighbor.X, neighbor.Y] == "JB" || grid[neighbor.X, neighbor.Y] == "JG" || grid[neighbor.X, neighbor.Y] == "JR")

307 var jewel = Jewels.First(j => j.X == neighbor.X && j.Y == neighbor.Y);
308 jewelsLocal.Remove(jewel);
309 }

311 grid[neighbor.X, neighbor.Y] = "oo";
312 queue.Enqueue(neighbor);
313 }
314 }

316 grid[node.X, node.Y] = "xx";

318 }
