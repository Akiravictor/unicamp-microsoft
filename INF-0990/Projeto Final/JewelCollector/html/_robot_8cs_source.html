234 private bool CollectJewel(Map map)
235 {
236 Jewel? jewel = null;
237
238 if (JewelUp(map))
239 {
240 jewel = map.Jewels.First(jewel => jewel.X == X - 1 && jewel.Y == Y);
241 }
242 else if (JewelDown(map))
243 {
244 jewel = map.Jewels.First(jewel => jewel.X == X + 1 && jewel.Y == Y);
245 }
246 else if (JewelLeft(map))
247 {
248 jewel = map.Jewels.First(jewel => jewel.X == X && jewel.Y == Y - 1);
249 }
250 else if (JewelRight(map))
251 {
252 jewel = map.Jewels.First(jewel => jewel.X == X && jewel.Y == Y + 1);
253 }
254
255 if (jewel != null)
256 {
257 if (jewel.GetType() == typeof(Blue))
258 {
259 var blue = (Blue)jewel;
260 RechargeEnergy(blue.RechargeEnergy());
261 }
262
263 Bag.AddItem(jewel.Value);
264 map.RemoveJewel(jewel);
265
266 return true;
267 }
268
269 return false;
270 }
